CREATE TABLE stock_user.stocks_part (
    id          NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    symbol      VARCHAR2(10) NOT NULL,
    trade_date  DATE NOT NULL,
    open        NUMBER(10,4),
    low         NUMBER(10,4),
    high        NUMBER(10,4),
    close       NUMBER(10,4),
    volume      NUMBER(19),
    CONSTRAINT stocks_symbol_date_unique UNIQUE (symbol, trade_date)
)
PARTITION BY RANGE (trade_date)
INTERVAL (NUMTOYMINTERVAL(3,'MONTH'))
(
    PARTITION p_base VALUES LESS THAN (DATE '1970-01-01')
)
TABLESPACE stocks_data;
